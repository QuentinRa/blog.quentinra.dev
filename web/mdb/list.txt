<script>
	$('#zoom').on('show.bs.modal', function (e) {
		let div = e.relatedTarget;
		let img = div.getElementsByTagName("img")[0];
		let modalTitle = document.getElementById('zoomLabel');
		let modalContent = document.getElementById('zoomBody');
		modalTitle.innerText = img.alt;
		modalContent.innerHTML = "<img src='"+img.src+"' alt='"+img.src+"' class='mw-100' />";
	});
</script>
